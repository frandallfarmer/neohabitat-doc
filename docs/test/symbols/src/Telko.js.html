1 /**
  2  * Telko: A brain-dead simple telnet-style relay meant to be used for
  3  * testing Elko servers with simple text JSON messages, usually pasted
  4  * in from a text file.
  5  * 
  6  * The problem is that some references change between executions, so 
  7  * statically stored (text file) JSON messages can't work
  8  * to refer to objects that are Elko 'clone's or users.
  9  * 
 10  * This file fixes that by allowing partial unique identifiers in the
 11  * "to:" field and attempts to promote them to fully qualified refs.
 12  * 
 13  * Any message may include a "Telko" object which allows the input 
 14  * to customize the session - current editable variables include:
 15  * Telko.logtime (boolean) which includes a timestamp in the log
 16  * Telko.delay (number) seconds to delay before the next message
 17  * is sent (useful only when redirecting STDIN from a file.)
 18  * 
 19  * Author: Randy Farmer 12/22/2016 
 20  */

 86 const Millis	 = 1000;
 87 /** Telko allows a parameter to delay the time between outgoing events (a crude simulation of a slow client: timeLastSent tracks events to be scheduled in the future even though they arrive on stdin instantly. */
 88 var	  timeLastSent = new Date().getTime();

100 /** Short names for all of the objects seen and the most recent fully qualified ref for that object. */
101 var Names = {};

103 /** An instance of all the objects seen during this session. Used for embedded variable substitution */
104 var History = {};

106 function pad0(n,z) {
107 	z = z || 1;
108 	return ("000" + n).slice(-(z + 1));
109 }
110 /**
111  * Returns a string to prefix log entries with. Uses Telko.logtime as a flag to return the result with, or wihout a timestamp.
112  * 
113  * @param String ptr The "pointer" string - put after the timestamp or at the head of the line depending on Telko.logtime
114  * @returns String The header for the log line...
115  */
116 function timestamp(ptr) {
117 	if (!Telko.logtime) { return ptr + " "; }
118 	var time = new Date();
119 	return	time.getFullYear()				+ "/" +
120 			pad0(time.getMonth() + 1)		+ "/" +
121 			pad0(time.getDate())			+ " " +		   
122 			pad0(time.getHours())			+ ":" +
123 			pad0(time.getMinutes())			+ ":" +
124 			pad0(time.getSeconds())			+ "." +
125 			pad0(time.getMilliseconds(), 3)	+ " " +
126 			ptr + " ";
127 }

129 /**
130  * A very aggressive name mapper. Splits on dashes '-' and dots '.'
131  * 
132  * @param String s The label to split up and attach to the names table
133  */
134 
135 function addName(s) {
136 	s.split("-").forEach(function(dash) {
137 		Names[dash] = s;
138 		dash.split(".").forEach(function(dot) {
139 			Names[dot] = s;
140 		});
141 	});
142 }

146 /**
147  * JSON parse the message, handling errors.
148  * 
149  * @param String s
150  * @returns Object The JSON object from the message or {}
151  */
152 function parseElko(s) {
153 	var o = {};
154 	try {
155 		o = JSON.parse(s);
156 	} catch (e) {
157 		Trace.warn("Unable to parse: " + s + "\n\n" + JSON.stringify(e, null, 2));
158 	}
159 	return o;
160 }
161 
162 /**
163  * 
164  * @param String s The message to be scanned for references ('ref's)
165  */
166 function scanForRefs(s) {
167 	var o = parseElko(s);
168 	if (o.to) {
169 		addName(o.to);
170 	}
171 	if (o.op && o.op === "make") {
172 		var ref = o.obj.ref;
173 		addName(ref);
174 		History[ref] = o;
175 		if (o.you) {
176 			var split	= ref.split("-");
177 			Names.ME 	= ref;
178 			Names.USER	= split[0] + "-" + split[1];
179 		}
180 	}	
181 }
182 
183 /**
184  * This allows the input (stdin/file) to use short-names for addressing objects. "randy" instead of "user-randy-13491283092"
185  * 
186  * @param String s
187  * @returns Either s or the value of Name[s], if it exists.
188  */
189 function substituteName(s) {
190 	return Names[s] || s;
191 }
192 
193 /**
194  * Telko supports a special state substitution. Any string that starts with "$" will trigger a lookup of the 
195  * state via the Names table. Example "$randy.obj.mod[0].x" will lookup "randy"'s formal ref in the $Names
196  * table, then the value of History.user-randy-1230958410291.obj.mod[0].x will be substituted. All substitutions will
197  * occur in place.
198  * 
199  * @param JSON Object m The object/message that will have it's parameters ($) substituted.
200  */
201 function substituteState(m) {
202 	for (var name in m) {
203 		if(m.hasOwnProperty(name)) {
204 			var prop = m[name];
205 			if ((typeof prop === 'string' || prop instanceof String) && prop.indexOf('$') !== -1) {
206 				var chunks = prop.split("$");
207 				for (var i = 1; i < chunks.length; i++) {
208 					var value  = chunks[i];
209 					var keys   = chunks[i].split('.');
210 					var first  = true;
211 					var obj;
212 					var mod;
213 					for(var j = 0; j < keys.length; j++) {
214 						var varseg = keys[j];
215 						if (first) {
216 							value = History[substituteName(varseg)];
217 							if (undefined === value) {	// No matching object, so substitute the key's value
218 								value = Names[varseg] || chunks[i];
219 								break;
220 							}
221 							if (undefined !== value.obj) {
222 								obj = value.obj;
223 								if (undefined !== obj.mods & obj.mods.length === 1) {
224 									mod = obj.mods[0];
225 								}
226 							}
227 							first = false;
228 						} else {
229 							value = (undefined !== mod && undefined !== mod[varseg]) ? mod[varseg] :
230 								(undefined !== obj && undefined !== obj[varseg]) ? obj[varseg] :
231 									value[varseg];
232 						}
233 					}
234 					chunks[i] = value;
235 				}
236 				if (chunks.length === 2 && chunks[0] === "") {
237 					m[name] = chunks[1];		// This preserves integer types, which have no leading chars
238 				} else {
239 					m[name] = chunks.join("");	// For in-string substitutions. 
240 				}
241 			}
242 		}
243 	}
244 }

247 /**
248  * This is the server object created with Net.connect. 
249  */
250 const Server = Net.connect(Telko.port, Telko.host, function() {
251 	Trace.info("Server connected @" + Server.address().address + ':'+ Server.address().port);
252 });
253 
254 /** 
255  * All we want to do with incoming packets is log them and register any incoming references for later lookup.
256  * 
257  * @param String s The packed coming in from the server.
258  * @returns
259  */
260 function processElkoPacket(s) {
261 	console.log(timestamp("<-") + s.trim());
262 	scanForRefs(s);
263 }
264 
265 /**
266  * Server.on deals with incoming data from the Server. We need to use Elko packet framing,
267  * inherited from HTML days - so as we find the end of each server packet, we process it.
268  */
269 Server.on('data', function(buf) {
270 	var framed = false;
271 	var firstEOL = false;
272 	var JSONFrame = "";
273 	var blob = buf.toString();
274 	for (var i=0; i < blob.length; i++) {
275 		var c = blob.charCodeAt(i);
276 		if (framed) {
277 			JSONFrame += String.fromCharCode(c);
278 			if (10 === c) {
279 				if (!firstEOL) {
280 					firstEOL = true;
281 				} else {
282 					processElkoPacket(JSONFrame);
283 					framed    = false;
284 					firstEOL  = false;
285 					JSONFrame = "";
286 				}
287 			}
288 		} else {
289 			if (123 === c) {
290 				framed = true;
291 				firstEOL = false;
292 				JSONFrame = "{";
293 			} else {
294 				if (10 !== c) {
295 					Trace.debug("IGNORED: " + c);					
296 				}
297 			}
298 		}
299 	}
300 	if (framed) {	
301 		processElkoPacket(JSONFrame);
302 		framed    = false;
303 		firstEOL  = false;
304 		JSONFrame = "";
305 	}
306 });

308 /**
309  * Server.on: What if the Elko Server breaks the connection? For now we tear the bridge down also.
310  * If we ever bring up a "director" based service, this will need to change on context changes.
311  */
312 Server.on('end', function() {
313 	Trace.info('Server disconnected...');
314 	process.exit(0);
315 });

317 /**
318  * Make last minute substitutions, and send a string version of the object to the server.
319  * 
320  * @param JSONObject obj The JSON Object/Message to send after substitution. 
321  */
322 function sendWithSubstituions(obj) {
323 	if (obj.to) {
324 		obj.to = substituteName(obj.to);
325 	}
326 	substituteState(obj);
327 	if (undefined !== obj.op && "entercontext" === obj.op && undefined === obj.context) {
328 		obj.context = Telko.context;
329 	}
330 	var msg = JSON.stringify(obj);
331 	console.log(timestamp("->") + msg.trim());
332 	Server.write(msg);
333 }

335 function startReadingSTDIN() {
336 	Readline.createInterface({input: process.stdin}).on('line', (input) => { handleInputLine(input); });
337 }
338 
339 function handleInputLine(input) {
340 	if (input.indexOf("{") === 0) {
341 		var obj = parseElko(input.toString());
342 		// Send the message, potentially in the future...
343 		var now  = new Date().getTime();
344 		var when = timeLastSent + Telko.delay;
345 
346 		// Any message may contain features the reconfigure it.
347 		if (obj.Telko) {
348 			Telko.delay = ( obj.Telko.delay || Telko.delay / Millis) * Millis;	// Effects NEXT message, not this one.
349 			Telko.logtime = (undefined !== obj.Telko.logtime) ? obj.Telko.logtime : Telko.logtime;
350 			// No support for changing servers at the moment...
351 			if (obj.Telko.host || obj.Telko.port) {
352 				Trace.error("No support for changing servers at the moment...");
353 			}
354 			delete obj.Telko;
355 		}
356 		
357 		if (when <= now) {
358 			sendWithSubstituions(obj);
359 			timeLastSent = now;
360 		} else {
361 			var delay = Math.max(0, (when - now));
362 			setTimeout(function () { sendWithSubstituions(obj); }, delay);
363 			timeLastSent = when;
364 		}
365 	} else if (input.indexOf("<") === 0) {
366 		var file = input.trim().slice(1);
367 		if ("STDIN" === file) {
368 			// Read from STDIN (usually triggered in a script, so that the timing works out...)
369 			startReadingSTDIN();
370 		} else {
371 			// Run a telko script file!
372 			executeScript(file);
373 		}
374 	}
375 }

377 function executeScript(script) {
378 	var contents;
379 	try {
380 		contents = File.readFileSync(script);
381 	} catch (err) {
382 		try {
383 			contents = File.readFileSync(script + ".elko");
384 		} catch (err) {
385 			Trace.error("Unable to read " + script + "[.elko]");
386 		}
387 	}
388 	if (contents) {
389 		var lines = contents.toString().split('\n');
390 		for (var i = 0; i < lines.length; i++) {
391 			handleInputLine(lines[i]);
392 		}
393 	}
394 }
395 
396 /** 
397  * Telko will take its input from either STDIN or as a list of script files using the command line paramter --files=FILE1,FILE2,...
398  */
399 if (Argv.files) {
400 	var files = Argv.files.split(',');
401 	for (var i = 0; i < files.length; i++) {
402 		executeScript(files[i]);
403 	}
404 	executeScript(Telko.end);
405 } else {
406 	startReadingSTDIN();
407 }
