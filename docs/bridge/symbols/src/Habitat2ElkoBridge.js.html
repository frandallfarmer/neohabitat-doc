 71 String.prototype.getBytes = function () {
 72 	var bytes = [];
 73 	for (var i = 0; i < this.length; ++i) {
 74 		bytes.push(this.charCodeAt(i));
 75 	}
 76 	return bytes;
 77 };

 79 /**
 80  * These are byte packets, and you needed to make sure to escape your terminator/unsendables.
 81  *  
 82  * @param b {buffer} The characters in the message to be escaped
 83  * @returns encoded char array
 84  * 
 85  */
 86 function escape(b, zero) {
 87 	zero = zero || false;
 88 	var r = [];
 89 	for (var i = 0; i < b.length; i++) {
 90 		var c = b[i];
 91 		if (c === HCode.END_OF_MESSAGE || c === HCode.ESCAPE_CHAR || (zero && c === 0)) {
 92 			r[r.length] = HCode.ESCAPE_CHAR;
 93 			c ^= HCode.ESCAPE_XOR;
 94 		}
 95 		r[r.length] = c;
 96 	}
 97 	return r;
 98 }

100 /**
101  * These were byte packets, and you needed to make sure to escape your terminator/unsendables.
102  *  
103  * @param b {buffer} The characters in the message to be escaped
104  * @returns decoded char array
105  */
106 function descape(b, skip) {
107 	var r = [];
108 	var i = skip || 0;
109 	while (i < b.length) {
110 		var c = b[i];
111 		if (c === HCode.ESCAPE_CHAR) {
112 			i++;
113 			c = b[i] ^ HCode.ESCAPE_XOR;
114 		}
115 		r[r.length] = c;
116 		i++;
117 	}
118 	return r;
119 }

124 function createServerConnection(port, host, client) {
125 
126 	var server = Net.connect({port: port, host:host}, function() {
127 		Trace.debug( "Connecting: " + 
128 				client.address().address +':'+ client.address().port +
129 				" <-> " +
130 				server.address().address + ':'+ server.address().port);
131 
132 		server.on('data', function(data) {
133 			var reset = false;
134 			try {
135 				reset = processIncomingElkoBlob(client, server, data);				
136 			} catch(err) {
137 				Trace.error("\n\n\nServer input processing error captured:\n" +
138 							JSON.stringify(err,null,2) +
139 							"\n...resuming...\n");
140 			}
141 			if (reset) {				// This connection has been replaced due to context change.
142 				Trace.debug("Destroying connection: " + server.address().address + ':' + server.address().port);
143 				server.destroy();
144 			}
145 		});
146 
147 		// What if the Elko server breaks the connection? For now we tear the bridge down also.
148 		// If we ever bring up a "director" based service, this will need to change on context changes.
149 
150 		server.on('end', function() {
151 			Trace.debug('Elko port disconnected...');
152 			if (client) {
153 				Trace.debug("Bridge being shutdown...");
154 				client.end();
155 			}
156 		});
157 		
158 		client.removeAllListeners('data').on('data', function(data) {
159 			try {
160 				parseIncomingHabitatClientMessage(client, server, data);				
161 			} catch(err) {
162 				Trace.error("\n\n\nClient input processing error captured:\n" + 
163 							JSON.stringify(err,null,2) +
164 							"\n...resuming...\n");
165 			}
166 		});
167 
168 		client.removeAllListeners('close').on('close', function(data) {
169 			Trace.debug("Habitat client disconnected.");
170 			if (server) { server.end();}
171 		});
172 	});
173 }

179 function futureSend(connection, data) {
180 	var now  = new Date().getTime();
181 	var when = Math.ceil(connection.timeLastSent + connection.lastSentLen * 8 / Argv.rate * Millis);
182 	
183 	connection.lastSentLen = data.length;
184 	if (when <= now) {
185 		connection.write(data);
186 		connection.timeLastSent = now;
187 	} else {
188 		var delay = Math.ceil(Math.max(0, (when - now)));
189 		var msg = (isString(data)) ? data : Buffer.from(escape(data));
190 		setTimeout(function () { connection.rawWrite(msg); }, delay);
191 		connection.timeLastSent = when;
192 	}
193 }

195 function toHabitat(connection, data, split) {
196 	split = split || false;
197 	if (connection.json) {
198 		connection.write(JSON.stringify(data));
199 		connection.write(connection.frame);
200 	} else {
201 		var header = data.slice(0,4);
202 		if (split) {
203 			var payload = data.slice(4);
204 			for (var start = 0; start < payload.length; start += HCode.MAX_PACKET_SIZE) {
205 				var bytes = payload.slice(start);
206 				var size = Math.min(HCode.MAX_PACKET_SIZE, bytes.length);
207 				var seqbyte = header[1] & HCode.SPLIT_MASK;
208 				var bs = "";
209 				if (start === 0) {
210 					seqbyte |= HCode.SPLIT_START;
211 					bs += "START ";
212 				}
213 				seqbyte |= HCode.SPLIT_MIDDLE;
214 				bs += "MIDDLE ";
215 				if (size === bytes.length) {
216 					seqbyte |= HCode.SPLIT_END;
217 					bs += "END";
218 				}
219 				header[1] = seqbyte;
220 				futureSend(connection, connection.packetPrefix);
221 				futureSend(connection, header);
222 				futureSend(connection, bytes.slice(0, size));
223 				futureSend(connection, connection.frame);
224 			}
225 		} else {
226 			futureSend(connection, connection.packetPrefix);
227 			futureSend(connection, data);
228 			futureSend(connection, connection.frame);
229 		}
230 	}
231 }

234 function habitatPacketHeader(start, end, seq, noid, reqNum) {
235 	var r = [];
236 	r[0] = HCode.MICROCOSM_ID_BYTE;
237 	r[1] = (seq | (end   ? 0x80 : 0x00) | 0x40 | (start ? 0x20 : 0x00)) & HCode.BYTE_MASK;
238 	if (undefined !== noid)   {r[2] = noid   & HCode.BYTE_MASK; }
239 	if (undefined !== reqNum) {r[3] = reqNum & HCode.BYTE_MASK; }
240 	return r;
241 }

248 var HabBuf = function (start, end, seq, noid, reqNum) {
249 	this.data = [];
250 	if (undefined !== start) {
251 		this.data = this.data.concat(habitatPacketHeader(start, end, seq, noid, reqNum));
252 	}
253 	this.send = function (client, split) {
254 		Trace.debug(JSON.stringify(this.data) + " -> client (" + client.sessionName + ")");
255 		toHabitat(client, Buffer.from(this.data, 'binary'), split);
256 	};
257 	this.add  = function (val) {
258 		if (Array.isArray(val)) {
259 			this.data = this.data.concat(val);
260 		} else {
261 			this.data.push(val);
262 		}
263 	};
264 };

266 var ContentsVector = function (replySeq, noid, ref) {
267 	this.container		= new HabBuf();
268 	this.noids			= [];
269 	this.objects		= new HabBuf();
270 	this.containers		= {};
271 	this.containerRef   = ref;
272 	this.containerNoid  = noid;
273 	this.replySeq		= (undefined === replySeq) ? HCode.PHANTOM_REQUEST : replySeq;
274 	
275 	this.add = function (o) {
276 		var mod = o.obj.mods[0];
277 		if (undefined === this.containerRef) {
278 			this.containerRef  = o.to;
279 			this.containerNoid = mod.noid;
280 			this.containers[this.containerNoid] = this;
281 		}	
282 		if (mod.noid !== this.containerNoid) {
283 			habitatEncodeElkoModState(mod, o.container, this.containers[this.containerNoid].objects);
284 			this.containers[this.containerNoid].noids.push(mod.noid, HCode.CLASSES[mod.type]);
285 		} else {
286 			habitatEncodeElkoModState(mod, o.container, this.containers[this.containerNoid].container);
287 		}
288 	};
289 	this.send = function (client) {		
290 		var buf = new HabBuf(
291 				true,
292 				true,
293 				this.replySeq,
294 				HCode.REGION_NOID,
295 				HCode.MESSAGE_DESCRIBE);
296 		this.container.data[4] = client.state.avatar.noid;	// TODO Remove this little magic
297 		buf.add(this.container.data);
298 		buf.add(this.noids);
299 		buf.add(0);
300 		buf.add(this.objects.data);
301 		buf.add(0);
302 		buf.send(client, true);
303 	};
304 };

322 /**
323  * 
324  * @param client
325  * @param server
326  * @param data
327  */
328 function parseIncomingHabitatClientMessage(client, server, data) {
329 	var send = data.toString().trim();
330 
331 	// Handle new connections - determine the protocol/device type and setup environment
332 
333 	if (undefined === client.json) {
334 		initializeClientState(client);
335 		var curly = send.indexOf("{");
336 		var colon = send.indexOf(":");
337 		if (curly !== -1 && curly < colon) {
338 			client.json			= true;
339 			client.binary		= false;
340 			client.frame    	= "\n\n";		
341 		} else if (colon !== -1) {			// Hacked Qlink bridge doesn't send QLink header, but a user-string instead.
342 			client.packetPrefix	= send.substring(0, colon + 1);
343 			client.json 		= false;
344 			client.binary		= true;
345 			client.frame		= String.fromCharCode(HCode.END_OF_MESSAGE);
346 			// overload write function to do handle escape! 			
347 			client.rawWrite = client.write;
348 			client.write = function(msg) {
349 				if (isString(msg)) {
350 					client.rawWrite(msg);
351 				} else {
352 					client.rawWrite(Buffer.from(escape(msg)));
353 				}
354 			};
355 			var longUsername = send.substring(0, colon);
356 			if (undefined !== UserDB[longUsername]) {
357 				client.userRef   		= UserDB[longUsername].userRef;	 // Lookup user name from comm key
358 			} else {
359 				client.userRef 			= "user-" + longUsername.toLowerCase();
360 			}
361 			Trace.debug(client.sessionName + " (Habitat Client) connected.");
362 		}
363 	}
364 
365 	// Unpack the message and deal with any special protocol transformations.
366 
367 	if (client.json) {
368 
369 		if (send.indexOf("{") < 0) { return; }	// Empty JSON text frame ignore without warning.
370 
371 		var o = {};
372 		try {
373 			o = JSON.parse(send);
374 			if (o && o.op) {
375 				if (o.op === "entercontext") {
376 					Trace.debug(o.user + " is trying to enter region-context " + o.context);					
377 				}
378 			}
379 		} catch (e) {
380 			Trace.warn("JSON.parse faiure client (" + client.sessionName + ") ->  Ignoring: " + JSON.stringify(send) + "\n" + JSON.stringify(e));
381 			return;
382 		}
383 		Trace.debug(client.sessionName + " -> " + JSON.stringify(o) + " -> server ");
384 		toElko(server, send);
385 	} else if (client.binary) {
386 		parseHabitatClientMessage(client, server, data);
387 	} else {
388 		Trace.debug("client (" + client.sessionName + ") -> Garbage message arrived before protocol resolution. Ignoring: " + JSON.stringify(data));
389 		return;
390 	}	
391 }

393 function parseHabitatClientMessage(client, server, data) {
394 	var hMsg    = descape(data.getBytes(), client.packetPrefix.length + 8);
395 	var seq 	= hMsg[1] & 0x0F;
396 	var end 	= ((hMsg[1] & 0x80) === 0x80);
397 	var start	= ((hMsg[1] & 0x20) === 0x20);
398 	var noid	= hMsg[2] || 0;
399 	var reqNum	= hMsg[3] || 0;
400 	var args	= hMsg.slice(4);
401 	var msg;
402 	
403 	Trace.debug("client (" + client.sessionName + ") ->  [noid:" + noid +
404 			" request:" + reqNum + " ... " + JSON.stringify(args) + "]");
405 
406 	if (undefined === client.connected) {
407 		client.state.who = client.packetPrefix;
408 		client.connected = true;
409 		// SHORT CIRCUIT: Direct reply to client without server... It's too early to use this bridge at the object level.
410 		var aliveReply = new HabBuf(true, true, HCode.PHANTOM_REQUEST, HCode.REGION_NOID, HCode.MESSAGE_IM_ALIVE);
411 		aliveReply.add(1  /* SUCCESS */);
412 		aliveReply.add(48 /* "0" */);
413 		aliveReply.add("BAD DISK".getBytes());
414 		aliveReply.send(client);
415 		return;	
416 	} else {
417 		if (seq !== HCode.PHANTOM_REQUEST) {
418 			client.state.replySeq = seq;	// Save sequence number sent by the client for use with any reply.
419 		}
420 
421 		if (noid === HCode.REGION_NOID && reqNum === HCode.MESSAGE_DESCRIBE) {
422 			// Transform MESSAGE_DESCRIBE to entering a context. Probably not right. FRF TODO
423 			var replySeq = (undefined === client.state.nextRegion) ?
424 									HCode.PHANTOM_REQUEST : client.state.replySeq;
425 			var changeContextMessage =	{
426 						to:			"session",	
427 						op:			"entercontext",
428 						context:	client.state.nextRegion || Argv.context,
429 						user:		client.userRef
430 					};
431 			Trace.debug("Sending 'entercontext' to " + changeContextMessage.context  +" on behalf of the Habitat client.");
432 			toElko(server, JSON.stringify(changeContextMessage));
433 			initializeClientState(client, client.userRef, replySeq);
434 			return;
435 			
436 		} else {
437 			var o	  	= client.state.objects[noid];
438 			var op	  	= (undefined === o.clientMessages[reqNum]) ? "UNSUPPORTED" : o.clientMessages[reqNum].op;
439 			var ref  	= o.ref;
440 			msg   		= {"to":ref, "op":op};   // Default Elko-Habitat message header
441 			
442 			if ("UNSUPPORTED" === op) {
443 				Trace.warn("*** Unsupported client message " + reqNum + " for " + ref + ". ***");
444 				return;
445 			}
446 
447 			if (undefined !== HCode.translate[op]) {
448 				client.state.replyEncoder = HCode.translate[op].toClient;
449 				if (undefined !== HCode.translate[op].toServer) {
450 					HCode.translate[op].toServer(args, msg);
451 				}
452 			}
453 		}
454 	}
455 	if (msg) {
456 		toElko(server, JSON.stringify(msg));
457 		Trace.debug(JSON.stringify(msg) + " -> server (" + client.sessionName + ")");
458 	}
459 	return;
460 }

462 function removeNoidFromClient(client, noid) {
463 	var o = client.state.objects[noid];
464 	var buf = new HabBuf(
465 			true,
466 			true,
467 			HCode.PHANTOM_REQUEST,
468 			HCode.REGION_NOID,
469 			HCode.MESSAGE_GOAWAY);
470 	buf.add(noid);
471 	buf.send(client);
472 	
473 	delete client.state.refToNoid[o.ref];
474 	delete client.state.objects[noid];
475 	if (o.className === "Avatar") {
476 		client.state.numAvatars--;
477 	}

586 function parseIncomingElkoServerMessage(client, server, data) {
587 	var o = {};
588 
589 	try {
590 		o = JSON.parse(data);
591 	} catch (e) {
592 		Trace.warn("JSON.parse faiure server (" + client.sessionName + ") ->  Ignoring: " + JSON.stringify(data) + "\n" + JSON.stringify(e));
593 		return;
594 	}
595 	
596 	if (o.to === "session") {
597 		if (o.op === "exit") {
598 			var reason = "Server forced exit [" + o.whycode + "] " + o.why;
599 			if (undefined !== client.avatarNoid && client.binary) {
600 				diagnosticMessage(client, reason, client.avatarNoid);
601 			}
602 			Trace.warn(reason);
603 		}
604 	}
605 
606 	if (o.op && o.op === "make" && o.you) {	// This connection's avatar has arrived - we have a habitat session!
607 		var name      = o.obj.name;
608 		var mod       = o.obj.mods[0];
609 		var regionRef = o.to.split("-");
610 		var userRef   = o.obj.ref.split("-");
611 		UserDB[name] = { regionRef: regionRef[0] + "-" + regionRef[1],
612 				         userRef:   userRef[0]   + "-" + userRef[1] };
613 		File.writeFile(UFILENAME, JSON.stringify(UserDB, null, 2));
614 		client.sessionName += ":" + name;
615 		client.avatarNoid   = mod.noid;
616 		if (client.json) { // We might have to tell the server that the avatar is visible - emulating C64 client behavior.
617 			toElko(server, JSON.stringify({ to:o.to, op:"FINGER_IN_QUE"}));
618 			toElko(server, JSON.stringify({ to:o.to, op:"I_AM_HERE"}));			
619 		}
620 	}
621 
622 
623 	if (o.type === "changeContext") {
624 		client.state.nextRegion = o.context;	// Save for MESSAGE_DESCRIBE to deal with later.
625 		createServerConnection(	client.port, 
626 								client.host,
627 								client);				// create a new connection for the new context 
628 		return true; // Signal this connection to die now that it's obsolete.
629 	}
630 
631 	// JSON client is just a relay...
632 	if (client.json) {
633 		Trace.debug("server (" + client.sessionName + ") -> "  + JSON.stringify(o) + " -> client (" + client.sessionName + ")");
634 		toHabitat(client, o);
635 		return;
636 	}
637 
638 	// NEXT UP, TRANSFORM ANY LOGIC
639 
640 	/* Mapping change region (choosing a canonical direction) to change context
641 	   is awkward. Habitat wants to send a NEWREGION command and a canonical
642 	   compass direction. Elko wants to respond to the request with permission
643 	   to set the user's context to the credentials it supplies, in effect telling
644 	   the client to "Ask me again to connect to such-and-such-a-place with these
645 	   credentials."
646 
647 	   I simply am having the bridge do the extra round trip on behalf of the
648 	   Habitat Client. 	
649 	 */
650 
651 	if (undefined === data || (undefined === o.op && undefined === o.type)) {
652 		Trace.warn("Badly formatted server message! Ignored: " + JSON.stringify(o));
653 		return;
654 	}
655 
656 	Trace.debug("server (" + client.sessionName + ") -> " + JSON.stringify(o));
657 
658 	/*	changeContext means that Elko wants the user to request a new context.
659 			The bridge will handle this, as this round-trip doesn't involve the 
660 		 	Habitat client. See the MESSAGE_DESCRIBE to see the followup... */
661 
662 	
663 	if (o.op === "ready") {
664 		// Eat this, since Elko thinks the region's done and the avatar will arrive later
665 		// Habitat wants the user's avatar as part of the contents vector.
666 		return;
667 	}
668 
669 	if (o.op === "delete") {
670 		removeNoidFromClient(client, client.state.refToNoid[o.to]);
671 		return;
672 	}
673 	
674 	if (o.op === "make") {
675 		var mod  = o.obj.mods[0];
676 		var noid = mod.noid || 0;
677 		o.noid			 = noid;
678 		o.ref			 = o.obj.ref;
679 		o.className		 = mod.type;
680 		o.classNumber	 = HCode.CLASSES[mod.type] || 0;
681 		if (undefined === HCode[mod.type]) {
682 			Trace.error("*** Attempted to instantiate class '" + o.className + "' which is not supported. Aborted make. ***");
683 			return;
684 		}
685 		o.clientMessages = HCode[mod.type].clientMessages;
686 		client.state.objects[noid] = o;
687 		client.state.refToNoid[o.ref] = noid;
688 		if (o.className === "Avatar") {
689 			client.state.numAvatars++;
690 		}
691 		if (client.state.waitingForAvatar) {
692 			if (o.you) {
693 				client.state.ref    	= o.obj.ref;
694 				client.state.region 	= o.to;
695 				client.state.avatar		= mod;				
696 				for (var i = 0; i < client.state.objects.length; i++) {
697 					if (undefined !== client.state.objects[i]) {
698 						client.state.contentsVector.add(client.state.objects[i]);
699 					}
700 				}
701 				Trace.debug(client.state.user + " known as object ref " + client.state.ref + " in region/context " + client.state.region + ".");
702 				client.state.contentsVector.send(client);
703 				client.state.contentsVector   =  new ContentsVector(); 		// May be used by HEREIS/makes after region arrival
704 				client.state.waitingForAvatar = false;
705 				if (client.state.numAvatars === 1) {
706 					var caughtUpMessage = new HabBuf(true, true, HCode.PHANTOM_REQUEST, HCode.REGION_NOID, HCode.MESSAGE_CAUGHT_UP);
707 					caughtUpMessage.add(1);			// TRUE
708 					caughtUpMessage.send(client);
709 				}
710 			}
711 		} else {
712 			Trace.debug("server (" + client.sessionName + ")  make -> HEREIS");
713 			var buf = new HabBuf(
714 					true,
715 					true,
716 					HCode.PHANTOM_REQUEST,
717 					HCode.REGION_NOID,
718 					HCode.MESSAGE_HEREIS);
719 			buf.add(o.noid);
720 			buf.add(o.classNumber);
721 			buf.add(0);
722 			habitatEncodeElkoModState(mod, o.container, buf);
723 			buf.add(0);
724 			buf.send(client, true);
725 		}
726 		return;
727 	}
728 
729 	// End of Special Cases - parse the reply/broadcast/neighbor/private message as a object-command.
730 
731 	if (o.type === "reply") {
732 		var buf = new HabBuf(true, true, client.state.replySeq, o.noid, o.filler);
733 		if (undefined !== client.state.replyEncoder) {
734 			client.state.replyEncoder(o, buf);
735 		}
736 		buf.send(client);
737 		return;
738 	}
739 	
740 	if (undefined !== HCode.SERVER_OPS[o.op]) {
741 		o.reqno		= HCode.SERVER_OPS[o.op].reqno;
742 		o.toClient	= HCode.SERVER_OPS[o.op].toClient;
743 		var buf = new HabBuf(true, true, HCode.PHANTOM_REQUEST, o.noid, o.reqno);
744 		if (undefined !== o.toClient) {
745 			o.toClient(o, buf);
746 		}
747 		buf.send(client);
748 		return;
749 	} else {
750 		Trace.warn("Message from server headed to binary client not yet converted. IGNORED:\n");
751 		return;
752 
753 	}
754 }

757 function processIncomingElkoBlob(client, server, data) {
758 	var framed = false;
759 	var firstEOL = false;
760 	var JSONFrame = "";
761 	var blob = data.toString();
762 	for (var i=0; i < blob.length; i++) {
763 		var c = blob.charCodeAt(i);
764 		if (framed) {
765 			JSONFrame += String.fromCharCode(c);
766 			if (10 === c) {
767 				if (!firstEOL) {
768 					firstEOL = true;
769 				} else {
770 					if (parseIncomingElkoServerMessage(client, server, JSONFrame)) {
771 						return true;		// Abort and pass along signal that this connection must reset.
772 					}
773 					framed = false;
774 					firstEOL = false;
775 					JSONFrame = "";
776 				}
777 			}
778 		} else {
779 			if (123 === c) {
780 				framed = true;
781 				firstEOL = false;
782 				JSONFrame = "{";
783 			} else {
784 				Trace.warn("IGNORED: " + c);
785 			}
786 		}
787 	}
788 	if (framed) {
789 		Trace.error("INCOMPLETE FRAME: " + JSONFrame);
790 	}
791 }
